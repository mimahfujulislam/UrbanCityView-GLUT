const float PI = 3.14159f;

void drawCircle(float cx, float cy, float r, int segments) {
    glBegin(GL_POLYGON);
    for (int i = 0; i < segments; i++) {
        float theta = 2.0f * PI * float(i) / float(segments);
        glVertex2f(cx + r * cosf(theta), cy + r * sinf(theta));
    }
    glEnd();
}

void display() {
    if (isNight) glClearColor(0.05, 0.05, 0.2, 1.0);
    else glClearColor(0.5, 0.8, 1.0, 1.0);

    glClear(GL_COLOR_BUFFER_BIT);

    SunMoon();
    DrawClouds();

    Building1();
    Building2();
    Building3();
    Building4();
    Building5();

    Tree1();
    Tree2();
    Tree3();
    Tree4();
    Tree5();

    RoadSystem();

    Car1();
    Car2();
    Car3();

    glutSwapBuffers();
}

void keyboard(unsigned char key, int x, int y) { //id17
    if (key == 'D' || key == 'd') isNight = false;
    if (key == 'N' || key == 'n') isNight = true;

    if (key == ' ') {
        if (moveCars || moveClouds) {
            moveCars = false;
            moveClouds = false;
        } else {
            moveCars = true;
            moveClouds = true;
        }
    }
    glutPostRedisplay();
}

void mouse(int button, int state, int x, int y) { //id18
    if (state == GLUT_DOWN) {
        if (button == GLUT_LEFT_BUTTON) moveCars = !moveCars;
        if (button == GLUT_RIGHT_BUTTON) moveClouds = !moveClouds;
    }
    glutPostRedisplay();
}

void timer(int val) { //id19
    if (moveCars) carMoveX += 0.5f;
    if (moveClouds) cloudMoveX += 0.2f;

    glutPostRedisplay();
    glutTimerFunc(16, timer, 0);
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_DOUBLE | GLUT_RGB);
    glutInitWindowSize(1200, 600);
    glutCreateWindow("Urban City View - Expanded Vertices");

    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    gluOrtho2D(0, 180, -35, 90);

    glutDisplayFunc(display);
    glutKeyboardFunc(keyboard);
    glutMouseFunc(mouse);
    glutTimerFunc(0, timer, 0);

    glutMainLoop();
    return 0;
}
